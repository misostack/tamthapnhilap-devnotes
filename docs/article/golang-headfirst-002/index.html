package main

import "fmt"

func main() {
	fmt.Println("Headfirst GO")
	fmt.Println("The alphabet")
	theAlphabet(true)
	theAlphabet(false)
}

func theAlphabet(uppercase bool) {
	var str = ""
	if uppercase {
		for i:= 'A'; i<= 'Z'; i++ {
			str += string(i) + " "
		}
	}else {
		for i:= 'a'; i<= 'z'; i++ {
			str += string(i) + " "
		}		
	}
	fmt.Println(str)
}

func theEvenNumbers(start int, end int) {
