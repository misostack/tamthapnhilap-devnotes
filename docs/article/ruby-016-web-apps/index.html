– Display objects with HTML

  • Refs : https://ruby-doc.org/stdlib-2.5.1/libdoc/erb/rdoc/ERB.html
$LOAD_PATH.unshift(File.join(File.dirname(__FILE__),'lib'))
require 'sinatra'
require 'todo'

get ('/') do	
	@paths = $LOAD_PATH
	erb :"todo/index"
end

<%# @paths %>
<% @paths.each do |path| %>
<li class="list-group-item"><%= path %></li>
<% end %>
